8pub fn vec_to_so3(omg: nalgebra::Vector3<f64>) -> nalgebra::Matrix3<f64> {
9    let so3mat = nalgebra::Matrix3::new(
10        0.0, -omg[2], omg[1], omg[2], 0.0, -omg[0], -omg[1], omg[0], 0.0,
11    );
12    so3mat
13}

20pub fn axis_ang3(expc3: nalgebra::Vector3<f64>) -> (nalgebra::Vector3<f64>, f64) {
21    let exp3d = nalgebra::DVector::from_row_slice(expc3.as_slice());
22    let omghatd = utils::normalize(exp3d);
23
24    let omghat = nalgebra::Vector3::new(omghatd[0], omghatd[1], omghatd[2]);
25    let theta = expc3.norm();
26
27    (omghat, theta)
28}

30pub fn matrix_exp3(so3mat: nalgebra::Matrix3<f64>) -> nalgebra::Matrix3<f64> {
31    let omgtheta = so3_to_vec(so3mat);
32    if utils::near_zero(omgtheta.norm()) {
33        nalgebra::Matrix3::identity()
34    } else {
35        let (_, theta) = axis_ang3(omgtheta);
36        let omgmat = so3mat / theta;
37        nalgebra::Matrix3::identity() + theta.sin() * omgmat + (1.0 - theta.cos()) * omgmat * omgmat
38    }
39}

41pub fn matrix_log3(r: nalgebra::Matrix3<f64>) -> nalgebra::Matrix3<f64> {
42    let acosinput = (r.trace() - 1.0) / 2.0;
43    if acosinput >= 1.0 {
44        nalgebra::Matrix3::zeros()
45    } else if acosinput <= -1.0 {
46        let omg;
47        if !utils::near_zero(1.0 + r[(2, 2)]) {
48            omg = (1.0 / (2.0 * (1.0 + r[(2, 2)]).sqrt()))
49                * nalgebra::vector![r[(0, 2)], r[(1, 2)], 1.0 + r[(2, 2)]];
50        } else if !utils::near_zero(1.0 + r[(1, 1)]) {
51            omg = (1.0 / (2.0 * (1.0 + r[(1, 1)]).sqrt()))
52                * nalgebra::vector![r[(0, 1)], 1.0 + r[(1, 1)], r[(2, 1)]];
53        } else {
54            omg = (1.0 / (2.0 * (1.0 + r[(0, 0)]).sqrt()))
55                * nalgebra::vector![1.0 + r[(0, 0)], r[(1, 2)], r[(2, 2)]];
56        }
57        vec_to_so3(omg)
58    } else {
59        let theta = acosinput.acos();
60        theta / 2.0 / theta.sin() * (r - r.transpose())
61    }
62}

64pub fn rp_to_trans(r: nalgebra::Matrix3<f64>, p: nalgebra::Vector3<f64>) -> nalgebra::Matrix4<f64> {
65    let bottom = nalgebra::matrix![0.0, 0.0, 0.0];
66    let one = nalgebra::vector![1.0];
67    nalgebra::stack![r, p; bottom, one]
